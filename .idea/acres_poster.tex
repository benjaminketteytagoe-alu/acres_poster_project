% ----- Block 1 -----
        \block{1.\ Problem, Context, and Users}{
            \takeawaybox{Modernising rental management to secure housing and financial transparency}
            \vspace{0.2em}
            Over \textbf{50\%} of Kigali residents rent, yet landlord--tenant relationships run on paper leases and handwritten receipts. Rwanda's rapid urbanisation is outpacing these systems, exposing landlords to revenue leakage and leaving tenants without proof of payment.
        }

% ----- Block 2 -----
        \block{2.\ Objectives and Scope}{
            \begin{itemize}
                \item \textbf{Aim:} Centralised web platform to digitise property management and lease compliance.
                \item \textbf{Obj. 1:} Multi-unit dashboard for financial and occupancy oversight.
                \item \textbf{Obj. 2:} Secure digital vault for leases with automated invoicing.
                \item \textbf{Obj. 3:} QR-code maintenance reporting requiring zero app installation.
                \item \textbf{In scope:} Lease tracking, payment logging, maintenance ticketing, role-based access.
                \item \textbf{Out of scope:} Bank API integration, IoT smart locks, multi-currency support.
            \end{itemize}
        }

% ----- Block 4 -----
        \block{4.\ Constraints and Design Decisions}{
            \begin{itemize}
                \item \textbf{Connectivity:} SSR pages optimised for low-bandwidth mobile networks.
                \item \textbf{Legal:} Complies with Rwanda's 2021 Data Protection Act; PII encrypted via bcrypt \& AES-256.
                \item \textbf{Literacy:} One-tap UI workflows designed for varied digital literacy.
                \item \textbf{Stack:} Next.js + Node.js + MySQL for SSR performance and data integrity.
                \item \textbf{Auth:} Unit-specific QR codes --- no tenant password required.
                \item \textbf{Trade-off:} Zero-install simplicity chosen over a richer native mobile app.
            \end{itemize}
        }

% ----- Block 5 -----
        \block{5.\ Proposed Solution and System Overview}{
            \takeawaybox{ACRES --- an Administrative Engine turning notebooks into a digital dashboard}

            \begin{minipage}[t]{0.48\linewidth}
                \textbf{Solution summary}\\[0.15em]
                {\small ACRES digitalises every landlord--tenant touchpoint. Landlords upload leases, log payments, and monitor occupancy from one dashboard. Tenants scan a door-mounted QR code to submit maintenance requests with no login or app needed. Alerts fire automatically for due rent, expiring leases, and unresolved tickets.}
            \end{minipage}
            \hfill
            \begin{minipage}[t]{0.48\linewidth}
                \textbf{Architecture overview}
                \begin{itemize}
                    \item \textbf{Frontend:} Next.js + Tailwind CSS (SSR)
                    \item \textbf{Backend:} Node.js REST API, JWT auth
                    \item \textbf{Database:} MySQL --- units, leases, payments
                    \item \textbf{Auth:} bcrypt hashing; QR-token tenant access
                    \item \textbf{Hosting:} Vercel (FE) + Railway (API + DB)
                \end{itemize}
            \end{minipage}

        \block{8.\ Impact, Contribution, and Learning}{
            \begin{itemize}
                \item \textbf{Technical:} QR-to-Web workflow enables zero-friction tenant access with full audit trails.
                \item \textbf{Users:} Landlords gain financial visibility; tenants get timestamped receipts and a maintenance pipeline.

% ----- Block 9 -----
        \block{9.\ Limitations and Next Steps}{
            \textbf{Limitations}
            \begin{itemize}
                \item No MTN MoMo / Airtel API --- payments verified manually.
                \item Assumes basic smartphone access and QR-scanning literacy.
                \item Small pilot (12 landlords); broader validation needed.
            \end{itemize}

            \vspace{0.3em}
            \textbf{Next steps}
            \begin{itemize}
                \item Integrate MTN MoMo \& Airtel Money for auto-reconciliation.
                \item Add SMS/USSD fallback for non-smartphone tenants.
                \item Expand to 50+ landlords across 5 Kigali districts.
                \item Explore RDB registration for formal property managers.
            \end{itemize}
        }
